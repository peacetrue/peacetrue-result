= 快速上手

== 引入类库

.成功类库，封装成功时返回的数据
[source%nowrap,maven]
----
<dependency>
  <groupId>com.github.peacetrue.result</groupId>
  <artifactId>peacetrue-result-success</artifactId>
  <version>1.0.0</version>
</dependency>
----

.异常支持类库，实现具体异常的转换
[source%nowrap,maven]
----
<dependency>
  <groupId>com.github.peacetrue.result</groupId>
  <artifactId>peacetrue-result-sample</artifactId>
  <version>1.0.0</version>
</dependency>
----
